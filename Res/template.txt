uniform float     iTime;uniform vec2      viewport;uniform mat4      prjInvMat;uniform mat4      mdlInvMat;vec3 yztozy(vec3 p){    return vec3(p.x,p.z,p.y);}vec2 coordToUV(vec2 coord,vec2 size){    return 2*coord/size-1;}void setupCamera(vec2 uv,out vec3 campos, out vec3 dir){     vec4 camdir = vec4(uv,1,1)*prjInvMat;     camdir = camdir/camdir.w;     vec4 camv = vec4(0,0,0,1);     camv = camv*mdlInvMat;     campos = camv.xyz;     vec4 dirp = camdir*mdlInvMat;     dir = normalize((dirp-camv).xyz);}void main() {    vec2 uv = coordToUV(gl_FragCoord.xy,viewport);	       vec3 pos,ray;    setupCamera(uv,pos,ray);    gl_FragColor = vec4(0,1,1,1);}
